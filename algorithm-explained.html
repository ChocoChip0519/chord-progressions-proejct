buildTransitionsFromSongs

songs.json의 실제 곡 데이터에서 코드 전이 확률 테이블을 자동 생성한다

실제 코드 (data.js)

function buildTransitionsFromSongs(songList, getUpgradeMap) {
  const counts = {};

  // 1단계: 모든 곡에서 [roman[i] → roman[i+1]] 전이 카운트
  for (const song of songList) {
    const map = getUpgradeMap ? getUpgradeMap(song) : null;
    const romans = map ? song.roman.map(r => map[r] ?? r) : song.roman;
    for (let i = 0; i < romans.length - 1; i++) {
      const from = romans[i];
      const to   = romans[i + 1];
      if (!counts[from]) counts[from] = {};
      counts[from][to] = (counts[from][to] || 0) + 1;
    }
  }

  // 2단계: 각 from 코드에서 출발하는 전이를 확률로 정규화
  const result = {};
  for (const from in counts) {
    const row   = counts[from];
    const total = Object.values(row).reduce((s, v) => s + v, 0);
    result[from] = {};
    for (const to in row) {
      result[from][to] = Math.round((row[to] / total) * 100) / 100;
    }
  }
  return result;
}

ChordGraph — 방향 가중치 그래프

인접 리스트로 구현. 각 엣지 가중치 = 전이 확률

실제 코드 (structures.js)

export class ChordGraph {
  constructor() { this.adj = new Map(); }

  loadFromData(data) {
    this.adj.clear();
    for (const from in data) {
      const m = new Map();
      for (const to in data[from]) m.set(to, data[from][to]);
      this.adj.set(from, m);
    }
  }

  // 현재 노드(from)의 이웃을 꺼내 diatonic 필터 후 상위 N개 반환
  getRecommendations(from, diatonic, topN = 4, genre = null) {
    const m = this.adj.get(from);
    if (!m) return [];
    let arr = [...m.entries()].map(([r, w]) => ({ romanNumeral: r, weight: w }));
    if (diatonic && diatonic.length) {
      const allowed = new Set(diatonic);
      const JAZZ_EXTRAS = new Set(["bII7", "VI7", "II7"]);
      arr = arr.filter(x =>
        allowed.has(x.romanNumeral) ||
        (genre === "rock" && x.romanNumeral === "bVII") ||
        (genre === "jazz" && JAZZ_EXTRAS.has(x.romanNumeral))
      );
    }
    arr.sort((a, b) => b.weight - a.weight);
    return arr.slice(0, topN);
  }

  // 가중치 기반 랜덤 워크 — 자동 코드 진행 생성용
  randomWalk(start, steps, diatonic, genre = null) {
    const out = [start];
    let cur = start;
    for (let i = 0; i < steps; i++) {
      const recs = this.getRecommendations(cur, diatonic, 8, genre);
      if (!recs.length) break;
      const total = recs.reduce((s, r) => s + r.weight, 0);
      let pick = Math.random() * total;  // 룰렛 휠 샘플링
      let chosen = recs[0].romanNumeral;
      for (const r of recs) {
        if (pick < r.weight) { chosen = r.romanNumeral; break; }
        pick -= r.weight;
      }
      out.push(chosen);
      cur = chosen;
    }
    return out;
  }
}

random(0, 1.0) → 범위에 해당하는 코드 선택. 가중치가 클수록 선택 확률이 높다.

자료구조 선택 이유 — 인접 리스트는 희소 그래프에 유리하다. 코드가 12개×2모드×다양한 7th 표기 = 수십 개 노드지만 각 노드의 평균 엣지는 4~6개에 불과하다. 인접 행렬(2D 배열)로 구현하면 대부분의 공간이 0으로 낭비되지만 인접 리스트는 실제 엣지만 저장한다.

Scoring

inferKey — 24후보 스코어링으로 키 자동 추론

코드 3개 이상 쌓이면 12키 × 2모드 = 24가지를 모두 채점한다

실제 코드 (music.js)

function inferKey(playedChords) {
  if (playedChords.length < 3) return [];

  const rootPcs  = playedChords.map(c => pc(c.rootNote)); // 0–11
  const qualities = playedChords.map(c => c.quality);

  const results = [];
  for (const keyName of NOTES) {           // 12키
    for (const mode of ["major", "minor"]) {  // 2모드
      const scPcs = scale(keyName, mode).map(pc);
      let score = 0;
      for (let i = 0; i < playedChords.length; i++) {
        // 루트음이 스케일에 속하면 +1
        if (scPcs.includes(rootPcs[i])) score += 1;
        // 토닉 코드가 quality도 맞으면 보너스 +0.5
        if (rootPcs[i] === pc(keyName)) {
          if (mode === "major" && ["maj","dom7","maj7"].includes(qualities[i])) score += 0.5;
          if (mode === "minor" && ["min","min7"].includes(qualities[i])) score += 0.5;
        }
      }
      results.push({ key: keyName, mode, confidence: score / playedChords.length });
    }
  }
  results.sort((a, b) => b.confidence - a.confidence);
  return results[0].confidence >= 0.7 ? [results[0]] : [];
}

ProgressionStack — Undo/Redo 이중 스택

current[], past[][], future[][]로 전체 히스토리를 관리한다

실제 코드 (structures.js)

export class ProgressionStack {
  constructor() {
    this.current = [];   // 현재 코드 진행
    this.past    = [];   // undo 히스토리 (스냅샷 배열)
    this.future  = [];   // redo 히스토리
  }

  push(item) {
    this.past.push([...this.current]);  // 현재 상태 스냅샷 저장
    this.current.push(item);
    this.future = [];                    // 새 행동 → redo 히스토리 초기화
  }

  undo() {
    if (!this.past.length) return null;
    this.future.push([...this.current]);
    this.current = this.past.pop();
    return this.current;
  }

  redo() {
    if (!this.future.length) return null;
    this.past.push([...this.current]);
    this.current = this.future.pop();
    return this.current;
  }

  set(newArr) { // 통째로 교체 (자동생성, 소급 업데이트)
    this.past.push([...this.current]);
    this.current = [...newArr];
    this.future = [];
  }

  load(newArr) { // 히스토리 없이 교체 (프로젝트 불러오기)
    this.past = []; this.current = [...newArr]; this.future = [];
  }
}

PlaybackQueue — FIFO 재생 큐

코드 진행을 순서대로 재생하기 위한 배열 기반 FIFO 큐

실제 코드 (structures.js)

export class PlaybackQueue {
  constructor() { this.items = []; }

  enqueueAll(items) { this.items.push(...items); }
  dequeue()          { return this.items.shift(); }
  isEmpty()          { return this.items.length === 0; }
  peek()             { return this.items[0] ?? null; }
  size()             { return this.items.length; }
  clear()            { this.items = []; }
}
